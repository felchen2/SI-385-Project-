Statistical Significance Testing

In[69]:
with_history = healthcare_data[healthcare_data['Family History'] == 1]
no_history = healthcare_data[healthcare_data['Family History'] == 0]

for var in ['Cholesterol', 'Triglycerides', 'Blood Pressure']:
    t_stat, p_val = ttest_ind(
        with_history[var],
        no_history[var],
        nan_policy='omit'
    )
    print(f"\nT-test for {var}:")
    print(f"t-statistic = {t_stat:.3f}, p-value = {p_val:.4f}")

Effect Size (Cohen's d)

In[70]:
def cohens_d(x, y):
    pooled_std = np.sqrt((np.std(x, ddof=1)**2 + np.std(y, ddof=1)**2) / 2)
    return (np.mean(x) - np.mean(y)) / pooled_std

for var in ['Cholesterol', 'Triglycerides', 'Blood Pressure']:
    d = cohens_d(with_history[var], no_history[var])
    print(f"Cohen's d for {var}: {d:.3f}")

Cohen's d for Cholesterol: 0.153
Cohen's d for Triglycerides: 0.133
Cohen's d for Blood Pressure: 0.208
